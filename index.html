function isPalindrome(line) {
    if (typeof(line) === "string") {
    return line === line.split("").reverse().join("");
  } else {
    let lineStr = String(line);
    let lineSplit = lineStr.split("");
    let lineReverse = lineSplit.reverse();
    let lineJoin = lineReverse.join("");
    let lineNum = Number(lineJoin);
    return line === lineNum;
    }
  }
